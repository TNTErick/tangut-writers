\newcommand{\charrow}[7]{% #1=row, #2=char, #3=pinyin, #4=Sofronov, #5=LFW's numbering, #6=four-corner, #7=meaning

  \pgfmathsetmacro{\pcen}{\yb+\cs+\ps/2}
  \pgfmathtruncatemacro{\lastC}{\COLS-1}

  \pgfmathsetmacro{\iho}{\yb+\ru*0.8}
  \pgfmathsetmacro{\iht}{\yb+\ru*0.6}
  \pgfmathsetmacro{\ihq}{\yb+\ru*0.4}
  \pgfmathsetmacro{\ihf}{\yb+\ru*0.2}

  % left info cell border
  \draw[black, line width=0.7pt]
    (0, \yb cm) rectangle (\li cm, \yb cm + \ru cm);

    % Sofronov

    at (\li cm - 0.12cm, \iho cm) {#4};

    at (0.12cm, \iht cm) {LFW};
